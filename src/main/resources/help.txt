// -- новые методы вместо save().
// -- помимо есть еще метод marge(). Пояснения: marge() - сохраняет сразу в бд не дожидаясь окончания сессии
// --                                           persist() - созроняет в кэш, сохронение в бд идет после окончания сессии
            session.persist(Object);
            session.persist(Object);
// -- новый метод вместо update()
            user.setRole(Role.USER);
            session.merge(Object);
// -- новый вместо метод delete()
            session.remove(Object);
//-- метод get() не поменялся
//-- возврощяет по первичному ключу можно создать пустой обьект с индификатором
            User userNew = session.get(Object.class, "Иван");
// -- во время сесии все измененимя в обьекте сохроняются
// -- но толька после метода get()
            userNew.setLastname("хуёв в жопу");
// -- закрывает транзакцию тем самым сохроняя обьект до комита
            session.flush();
// -- все хранится в кэше, и когда мы делаем session.clear();
// -- тем самым очищаем кэш и все наши операции до вызова
// -- этого метода не будут выполнены после .commit();
            session.clear();
// -- удоляет определенный обьект из кэша .evict(Object);
            session.evict(Object);
// -- обновит из базы данных текущий обьект данные в кеше и не сохранит(не работае при мнгновенном сохраниние пример: marge())
            session.refresh(Object);
// -- проверяет изменился ли обьект в кэше до окончания сесии, сравнивает с данными из бд.
            session.isDirty(Object);
